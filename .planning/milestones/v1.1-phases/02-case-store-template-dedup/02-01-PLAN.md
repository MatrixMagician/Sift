## Interfaces (contracts for tasks 2–3; implement exactly these signatures)

```python
# store.py
_RAW_ZSTD_THRESHOLD = 4096            # UTF-8 encoded bytes (STORE-02)
_MAX_RAW_BYTES = 128 * 2**20          # zstd-bomb cap (T-02-01)
def _encode_raw(raw: str) -> str | bytes: ...
def _decode_raw(value: str | bytes) -> str: ...
def _migration_2(conn: sqlite3.Connection) -> None: ...

@dataclass(frozen=True)
class TemplateGroup:
    template_id: str          # sha256(template)[:16], mirrors event_id idiom
    template: str             # masked message
    count: int
    first_ts: str | None      # ISO 8601 string or None
    last_ts: str | None
    severity_max: str         # one of the six-severity CHECK vocabulary
    exemplar_event_ids: list[str]

class CaseStore:
    def iter_event_summaries(self) -> Iterator[tuple[str, str | None, str, str]]:
        """Yield (event_id, ts, severity, message) in canonical order via cursor
        iteration — never fetchall, never selects raw."""
    def replace_template_groups(self, groups: Iterable[TemplateGroup]) -> None:
        """DELETE FROM template_groups then executemany insert. Caller owns the
        transaction."""
    def query_template_groups(self) -> list[TemplateGroup]:
        """ORDER BY count DESC, template ASC."""

# pipeline/dedup.py  (typer-free, print-free, SQL-free)
MASK_VERSION = 1
EXEMPLAR_K = 5
def mask(message: str) -> str: ...
def template_id(template: str) -> str: ...
def rebuild_template_groups(store: CaseStore) -> int:
    """Recompute all groups from the store; returns the group count. Wraps
    replace_template_groups + set_meta('mask_version', ...) in one
    store.transaction()."""
```
